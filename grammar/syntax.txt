<prgm> ::= <stmt>

<stmt> ::= "IF" <condi> "THEN" nl {<expr>} nl "ENDIF"
| "WHILE" <condi> "DO" nl {<expr>} nl "ENDWHILE"
| "INPUT" <ident>
| "OUTPUT" <expr | string>
| "LET" <ident> "=" <expr | string | bool> 
| ("INCR" | "DECR") <ident> (<expr>)?
| "FOR" <ident> "=" <expr> "TO" <expr> nl {<expr>} nl "NEXT" <ident>

<condi> ::= <expr> ("==" | ">" | ">=" | "<" | "<=" | "!=") <expr>
<comment> ::= "//" <string>+
<expr> ::= <term> {("-" | "+") <term>}
<term> ::= <factor> {("/" | "*") <factor>}
<factor> ::= <int> | <LPAREN> <expr> <RPAREN>


<nl> ::= '\n'+
<LPAREN> ::= '('
<RPAREN> ::= ')"
